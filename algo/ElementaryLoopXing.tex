\begin{algorithm}\label{algo_ElementaryLoopXing}
\footnotesize
\SetAlgoVlined
\SetKw{return}{return}\SetKw{Break}{break}
\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
\caption{析取逻辑程序基本环的近似识别算法$ElementaryLoop^*(L,P)$}
\Input{$L$ a loop of $P$，$P$ a disjunctive logic program}
\Output{if returns $L$, then $L$ may be a elementary loop\\otherwise, $L$ is not a elementary loop}

\For{each atom $a \in L$} {
	$G^*:=$ the $L\backslash \{a\}$ induced subgraph of $G_P$\\
	$SCC^*:=$ the set of SCCs of $G^*$\\
	\For{each $C \in SCC^*$} {
		\If{$R^-_L(C) \subseteq R^-(L)$} {
			\return $C$
		}
		\Else {
			$G_C:=$ the $C\backslash head(R_L^-(C)\backslash R^-(L))$ induced subgraph of $G^*$\\
			$SCC_C:=$ the set of SCCs of $G_C$\\
			append new elements from $SCC_C$ to $SCC^*$
		}
	}
}

\return $L$
\end{algorithm}
