\begin{algorithm}\label{algo_WeakElementaryLoopDLP3}
\footnotesize
\SetAlgoVlined
\SetKw{return}{return}\SetKw{Break}{break}
\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
\caption{判断程序是否属于HWEF*程序的算法$HWEF^*(P)$}
\Input{$P$ a disjunctive logic program}
\Output{if returns true, then $P$ is a HWEF program\\otherwise, $P$ may be a HWEF program}

$S:=\{\{a,b\}|$there is a rule $r\in P s.t. \{a,b\}\subseteq head(r)\}$\\
\For{each $E \in S$} {
	\If{$E$ is a weak elementary loop of $P$} {
		\return false
	}
	\If{$EWEL(P,E)$ returns true} {
		\return false
	}
}

\return true
\end{algorithm}
