\begin{algorithm}\label{algo_WeakElementaryLoopDLP3}
\footnotesize
\SetAlgoVlined
\SetKw{return}{return}\SetKw{Break}{break}
\SetKwInOut{Input}{输入}\SetKwInOut{Output}{输出}
\caption{判断程序是否属于HWEF*程序的算法$HWEF^*(P)$}
\Input{析取逻辑程序$P$}
\Output{若返回真，则是$HWEF$程序；若返回假，则可能是}

$S :=\{ \{ a, b \} |$存在规则$r \in P$，使得$\{ a, b \} \subseteq head(r)\}$\\
\For{原子集$E \in S$} {
	\If{$E$是$P$的基本环} {
		\return false
	}
	\If{$EWEL(P,E)$返回true} {
		\return false
	}
}

\return true
\end{algorithm}
