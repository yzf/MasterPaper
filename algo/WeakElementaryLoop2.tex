\begin{algorithm}\label{algo_WeakElementaryLoopDLP2}
\footnotesize
\SetAlgoVlined
\SetKw{return}{return}\SetKw{Break}{break}
\SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
\caption{判断是否不存在包含某环的弱基本环的算法$EWEL(P,E)$}
\Input{$P$ a disjunctive logic program，$E$ a loop of $P$}
\Output{if returns false, then there is no elementary loop $L\supset E$}

$G:=$ the positive dependency graph of $P$\\
\If{there does not exist a SCC $C$ of $G$ s.t. $E \subseteq C$} {
	\return false
}
$C:=$ the SCC of $G$ s.t. $E\subseteq C$\\
$R_E:=\lbrace r|r \in P$ and $body^+(r) \cap E \ne \emptyset \rbrace$\\
\For{each $r \in R_E$} {
	$G_r:=$ the $C \backslash (body^+(r) \backslash E)$ induced subgraph of $G$\\
	\If{there exists a SCC $C_r$ of $G_r$ s.t. $E \subseteq C_r$ and $head(r) \cap (C_r \backslash E) \ne \emptyset$} {
		\return true
	}
}
\return false
\end{algorithm}
